def compute_topological_charge(phase_field, center_point, radius_estimator='adaptive'):
    """
    Compute topological charge with adaptive radius selection.
    Parameters:
        radius_estimator: 'adaptive' (based on local coherence) or float
    Returns:
        charge: Integer topological charge
        circulation: Circulation value
    """
    if radius_estimator == 'adaptive':
        radius = estimate_vortex_scale(phase_field, center_point)
    else:
        radius = radius_estimator
    theta_path = np.linspace(0, 2*np.pi, 64, endpoint=False)
    x_path = center_point[0] + radius * np.cos(theta_path)
    y_path = center_point[1] + radius * np.sin(theta_path)
    phase_along_path = interpolate_phase(phase_field, x_path, y_path)
    circulation = np.trapz(np.gradient(phase_along_path))
    charge = int(np.round(circulation / (2 * np.pi)))
    return charge, circulation

class SurrogateGenerator:

    def volume_conduction_surrogate(self, source_data, head_model='bem'):
        """
        Apply boundary element model (BEM) for realistic volume conduction.
        Parameters:
            head_model: 'bem' or 'fem' (finite element model)
        """
        n_sources = source_data.shape[0]
        random_orientations = random_unit_vectors(n_sources)
        forward_model = load_bem_model(head_model)
        surrogate = forward_model @ (source_data * random_orientations)
        return surrogate
    
    def phase_randomized_surrogate(self, source_data):
        fft_data = np.fft.fft(source_data, axis=-1)
        random_phases = np.random.uniform(0, 2*np.pi, fft_data.shape)
        surrogate_fft = np.abs(fft_data) * np.exp(1j * random_phases)
        return np.real(np.fft.ifft(surrogate_fft, axis=-1))
